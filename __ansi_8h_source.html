00104 /*  The traditional meaning of 'extern inline' for GCC is not
00105   to emit the function body unless the address is explicitly
00106   taken.  However this behaviour is changing to match the C99
00107   standard, which uses 'extern inline' to indicate that the
00108   function body *must* be emitted.  Likewise, a function declared
00109   without either 'extern' or 'static' defaults to extern linkage
00110   (C99 6.2.2p5), and the compiler may choose whether to use the
00111   inline version or call the extern linkage version (6.7.4p6).
00112   If we are using GCC, but do not have the new behaviour, we need
00113   to use extern inline; if we are using a new GCC with the
00114   C99-compatible behaviour, or a non-GCC compiler (which we will
00115   have to hope is C99, since there is no other way to achieve the
00116   effect of omitting the function if it isn't referenced) we use
00117   'static inline', which c99 defines to mean more-or-less the same
00118   as the Gnu C 'extern inline'.  */
00119 #if defined(__GNUC__) && !defined(__GNUC_STDC_INLINE__)
00120 /* We're using GCC, but without the new C99-compatible behaviour.  */
00121 #define _ELIDABLE_INLINE extern __inline__ _ATTRIBUTE ((__always_inline__))
00122 #else
00123 /* We're using GCC in C99 mode, or an unknown compiler which
00124   we just have to hope obeys the C99 semantics of inline.  */
00125 #define _ELIDABLE_INLINE static __inline__
00126 #endif
00127 
00128 #if __GNUC_PREREQ (3, 1)
00129 #define _NOINLINE       __attribute__ ((__noinline__))
00130 #define _NOINLINE_STATIC    _NOINLINE static
00131 #else
00132 /* On non-GNU compilers and GCC prior to version 3.1 the compiler can't be
00133    trusted not to inline if it is static. */
00134 #define _NOINLINE
00135 #define _NOINLINE_STATIC
